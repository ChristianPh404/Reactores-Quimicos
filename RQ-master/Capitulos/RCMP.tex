\chapter{Reactor Continuo Mezcla Perfecta}
\section{Introducción}
\noindent nos vamos a encontrar con un reactor tipo tanque agitado, que el diseño del agitador va a permitir una homogeneización de la mezcla reaccionante, en un instante t tendremos la misma concentración y la temperatura en cualquier punto del reactor.
el balance global de materia para un componente i es:
\begin{equation}
	\frac{dM}{dt} = m_{0} - m_{s}
\end{equation}
como por la configuracion del reactor se trabaja a volumen constante y al trabajar en un intervalo que podemos considerar el valor de la densidad constante se puede trabajar con caudales volumetrico.
\begin{equation}
	\sum F_0 = \sum F_s
\end{equation}
el balance a componentes queda expresado como:
\begin{equation}
	\frac{dN_i}{dt} = n_{i0} - n_{is} + r_i V \rightarrow \frac{dC_i}{dt} = \frac{F}{V}(c_{i0}-c_{is})+r_i
\end{equation}
dando la evolución de la cooncentracion de cualquier componente en un tiempo t. para el balance entalpico seria:
\begin{equation}
	\frac{d}{dt}  (M\cdot C_p \cdot T) = m \cdot cp (T_0 - T_s) + V \cdot \sum_{k=1}^{h} \Delta H_k \cdot r_k + Q(T)
\end{equation}
que da a lugar de forma explicita a la siguiente ecuación:
\begin{equation}
	\frac{dT}{dt} = \frac{F}{V} \cdot (T_0 - T_s) +\sum_{k=1}^{h} \frac{\Delta H_k \cdot r_k}{\rho \cdot C_p} + \frac{K\cdot A\cdot (T_j-T)} {V\cdot \rho \cdot C_p}
\end{equation}
visto en el apartado de modelizacion general, la resolucion del modelo nos permitiria calcular la evolución de concentraciones y temperatura a lo largo del tiempo, ese sistema %*Introducir mas tarde la referencia

\noindent llegara al estado estacionario, los reactores continuos acaban siempre al estado estacionario, donde hay una conversión de salida constante y una temperatura constante .
los reactores continuos van a evolucionar desde su puesta en marcha al estado estacionario, donde las corrientes de entrada y salida equilibran al reactor.
\section{Isotermo}
\noindent Para una reacción unica con un reactivo limitante A con un balance a componente es suficente siendo las ecuaciones de diseño:

\begin{equation}
	\frac{dN_A}{dt} = n_{A0} - n_{As} - r_A V \xrightarrow{\text{e.e}} \frac{dN_A}{dt} = 0
\end{equation}
\nt{la velocidad es negativa debido a que es un reactivo y por tanto se consume}
\noindent Si ponemos esta ecuacion en función de la conversión:
\begin{equation}
	n_{AR} (1-X_{A0}) - n_{AR} (1-X_{AS}) - (r_A)_s\cdot V = 0
\end{equation}
\noindent ademas si agrupamos términos para simplificar la ecuación queda de la siguiente forma:
\begin{equation}
	n_{AR} (X_{AS}-X_{A0}) = (r_A)_s\cdot V
\end{equation}
y recordando que $\tau$, tiempo espacial, es la relacion entre el volumen del reactor y un caudal de referencia normalmente inicial, y por otro lado tambien se puede encontrar la definicion del tiempo de residencia que es el volumen del reactor entre el caudal de salida,
en este caso ambos coinciden ya que el reactor es continuo. En bioreactores tambien se usa la inversa del $\tau$ que es la velocidad de dilución.
si en el balance anterior introducimos el tiempo espacial se obtiene la siguiente expresión:
\begin{equation}
	\frac{\textcolor{blue}{n_{AR}} (X_{AS}-X_{A0})}{\textcolor{blue}{F}} = \frac{(r_A)_s\cdot \textcolor{darkgreen}{V}}{\textcolor{darkgreen}{F}} \rightarrow \textcolor{darkgreen}{\tau} = \frac{\textcolor{blue}{C_{AR}}(X_{AS}-X_{A0})}{(r_A)_s}
\end{equation}
que es la ecuación de diseño para un reactor continuo mezcla perfecta, es una expresión algebraica que nos permite calcular el tiempo espacial necesario en el reactor y que por tanto conociendo el caudal  de entrada nos permite calcular el volumen del reactor. como normalmente el estado de referencia es el inicial $X_{A0}=0$, $C_{AR}$ = $C_{A0}$, por tanto la ecuación queda de la siguiente forma:
\begin{equation}
	\textcolor{darkgreen}{\tau} = \frac{\textcolor{blue}{C_{AR}}X_{AS}}{(r_A)_s}
\end{equation}
\section{Adiabático}

\noindent Seguimos con reacción unica para poder analizar el sistema desde un punto de vista analitico, el modelo, donde el balance de materia es exactamente igual pero introducimos el balance entalpico,
\begin{equation}
	\underbrace{\frac{d}{dt}  (M\cdot C_p \cdot T)}_{\text{Termino de acumulación}} = \underbrace{m \cdot cp (T_0 - T_s)}_{\text{Termino de entalpías de entrada y salida}} + \underbrace{ \Delta H_A \cdot r_A \cdot V}_{\text{Termino de reacción}}
\end{equation}
siendo los balances de materia y energia en el estado estacionario los siguientes:
\begin{align}
	(r_A)_s V            & = \textcolor{red}{n_{AR} (X_{AS}-X_{A0})}
	                     & \quad                                     & \leftarrow \text{balance de materia} \\
	\Delta H_A (r_A)_s V & = m\, c_p (T_0 - T_s)
	                     & \quad                                     & \leftarrow \text{balance de energía}
\end{align}
donde nos planteamos lo siguiente: ¿Podemos funcionar en un reactor continuo de forma adiabatico?
Hay que tener en cuenta que supone que hemos aislado el reactor, el reactor esta aislado para no tener intercambio de energía con el exterior por tanto si el sistema es exotérmico vamos a tener un aumento de temperatura y si es endotérmico vamos a tener una disminución de temperatura, pero un reactor continuo debe llegar al estado estacionario, por tanto
ese aumento o disminución de temperatura debe ser contrarestado por la corriente de alimentación, es decir que la corriente de entrada sirva para enfriar o calentar el sistema, por tanto, como $T_s$ queremos tenerla fija, el aumento o disminucion de $\Delta H_a \cdot (r_A)_s \cdot V$ influira en la $T_0$ eso es un funcionamiento autotérmico.
\begin{equation}
	\Delta H_A\cdot \textcolor{red}{n_{AR} (X_{AS}-X_{A0})} = {m \cdot cp (T_0 - T_s)}
\end{equation}
y trabajando con dicha expresión se puede llegar a que la temperatura del reactor es una funcion lineal de la conversión:
\begin{align}
	T_s & = T_0 + \frac{\Delta H_A \cdot \textcolor{red}{n_{AR}}}{\textcolor{red}{m}\, c_p} (X_{AS}-X_{A0})
	    &                                                                                                                                                              & \text{\scriptsize donde }
	\textcolor{red}{n_{AR}} = \textcolor{red}{C_{AR} V},\;
	\textcolor{blue}{m} = \textcolor{blue}{\rho V}                                                                                                                                                 \\
	[6pt]
	T_s & = T_0 + \frac{\Delta H_A \cdot \textcolor{red}{C_{AR}}\cancel{\textcolor{red}{V}}}{\textcolor{blue}{\rho}\cancel{\textcolor{blue}{V}}\, c_p} (X_{AS}-X_{A0})
\end{align}
es decir vamos a tener un aumento/disminución lineal de la temperatura, que compensaremos con la corriente de entrada. Aunque este funcionamiento es muy limitado ya que ante cualquier perturbación en el medio no se puede actuar sobre ella al no existir un intercambiador de calor y por tanto se desestabiliza.
para el tiempo espacial esta la expresión descrita en la Ec. 3.10
\section{No isotermo}
\noindent De forma analoga al regimen adiabatico, el balance de materia es exactamente igual pero en el balance entalpico introduciendo el intercambiador de calor queda de la siguiente forma:
\begin{equation}
	\frac{d}{dt}  (M\cdot C_p \cdot T) = m \cdot cp (T_0 - T_s) + \Delta H_A \cdot r_A \cdot V +  U \cdot S \cdot (T_{j}-T)
\end{equation}
\nt{considerando esa expresión del intercambiador de calor debiado a que se mantiene la temperatura del sistema constante bien por que este en un cambio de fase (calor latente) o bien por que se trabajamos con un caudal tan alto que la temperatura apenas varia y se pueda considerar constante}

En estado estacionario, para ver las ecuaciones de diseño los balances quedan de la siguiente forma:

\dfn{Balances de materia y energía en estado estacionario}{
	\begin{align}
		(r_A)_s \cdot V
		 & = n_{AR} (X_{AS}-X_{A0})
		 &                                                         & \frac{dN_A}{dt}=0       \\[6pt]
		\Delta H_A \cdot (r_A)_s \cdot V
		 & = m \cdot c_p (T_s - T_0) + U \cdot S \cdot (T_{s}-T_j)
		 &                                                         & \frac{d(m C_p T)}{dt}=0
	\end{align}
}
\begin{flushleft}

	\vspace{1em}
	Se obtiene un \textbf{sistema de ecuaciones algebraicas no lineales} cuya resolución requiere, en general, \textbf{métodos iterativos}, ya que las incógnitas $T_s$ y $X_{AS}$ están acopladas. La no linealidad del sistema proviene de la \textbf{dependencia de la cinética con la temperatura}, descrita mediante la ecuación de Arrhenius, de carácter exponencial.
	\vspace{1em}
	Para que exista una solución física, deben satisfacerse simultáneamente el \textbf{balance de materia} y el \textbf{balance de energía}. Desde un punto de vista gráfico, esto equivale a representar ambas ecuaciones y buscar sus puntos de intersección. Dado que el sistema es no lineal, pueden existir \textbf{múltiples puntos de corte},
	lo que implica la posible \textbf{multiplicidad de estados estacionarios}. Si el sistema tiene un solo estado estacionario, llegara de una forma mas rapida o mas lenta pero si tiene varios, hay que estudiar cuales son las condiciones para llegar a uno o a otro. y uno de esos estados estacionarios sera el optimo, es decir el que nos de mayor conversion o mayor o bien es el unico posible por la resistencia de los materiales ya que la temperatura en otro estado estacionario sea critica.
\end{flushleft}
\subsection{Multiplicidad de estados estacionarios}
\begin{flushleft}
	Vamos a analizar ahora la posible \textbf{multiplicidad de estados estacionarios} en reactores continuos de mezcla perfecta.
	\vspace{0.5em}
	En estado estacionario, la temperatura y la conversión permanecen constantes, lo que fija los valores de los parámetros cinéticos y, en consecuencia, la velocidad de reacción y la conversión alcanzada en el reactor.
	\vspace{0.5em}
	Para que dicho estado estacionario sea posible, es necesario que el calor generado por la reacción se equilibre con el calor eliminado del sistema. Este último tiene lugar tanto por el intercambio de entalpía con las corrientes de entrada y salida como por el sistema de intercambio térmico.

	El término de generación de calor puede expresarse como:
\end{flushleft}
\begin{align}
	H_G & = \Delta H_A \cdot (r_A)_s \cdot V
	    &                                     & \quad \text{\scriptsize donde $H_G$ heat gain (calor ganado)}  \\
	H_L & = m c_p(T_s - T_0) + U S(T_{s}-T_j)
	    &                                     & \quad \text{\scriptsize donde $H_L$ heat lowed (calor cedido)}
\end{align}

y por el otro lado todos los terminos son conocidos queda de la siguiente forma:

\begin{equation}
	H_L = \underbrace{(m\cdot c_p+U\cdot S)}_{\text{pendiente}} T_s - \underbrace{(m\cdot c_p T_0 + U\cdot S \cdot T_j)}_{\text{ordenada en el origen}}
\end{equation}
\noindent por tanto si representasemos el calor ganado en el sistema por el calor dado por la entalpia del proceso y la linea recta que supone el calor modulado en el proceso, las curvas se cortarian, en un sistema endotermico tendria solo un punto de corte
pero en un sistema exotermico tendria desde 1 a 3 puntos de corte, por tanto en los sistemas \textbf{exotermicos} se puede dar multiplicidad de estados estacionarios. Siendo la mayoria de los procesos industriales exotermicos, por tanto es necesario tenerlo en cuenta, y la recta del calor modulado
se puede actuar sobre ella cambiando la temperatura de entrada o la del intercambiador de calor, cambiando asi su ordenada y por lo tanto llevarlo a tener un solo punto de corte, o que sea mas adecuado para el proceso. \begin{itemize}
	\item Caso I: Un solo estado estacionario a baja temperatura.
	\item Caso III: Un solo estado estacionario a alta temperatura.
	\item Caso II: tres estados estacionarios
	      \begin{itemize}
		      \item A: Estado estacionario conversión y temperatura baja
		      \item B: Estado estacionario inestable
		      \item C: Estado estacionario conversión y temperatura alta

\end{itemize}
\nt{ Puede ser necesario trabajar en estados estacionarios intermedios ya que el estado de alta conversión, no es facilmente regulable y puede dar problemas de otras reacciones secundarias no deseadas o de resistencia de materiales... por lo que no es conveniente.}
para entender la estabilidad de los estados estacionarios, la mejor manera es fijarnos en el caso II donde para el punto A hay baja temperatura por tanto bajo valor de la constante cinetica y baja conversión, si trabajabmos en el punto C y hay una perturbación en el sistema
y aumenta como se puede ver  el calor que modula(HL) es mayor que el calor generado(HG), por tanto la temperatura tenderá a disminuir, y volverá al estado estacionario, por tanto el estado estacionario C es estable,
la mejor forma de entender es que si se cumple:
\begin{equation}
	\frac{dH_G}{dT_s} < \frac{dH_L}{dT_s} \quad \text{\scriptsize Estado estable}
\end{equation}
en el caso \textbf{B}, si hay una perturbacion que aumente la temperatura como el calor generado es mayor que el calor eliminado la temperatura seguira aumentando hasta llegar al estado C, por el contrario si la perturbacion disminuye la temperatura como el calor eliminado es mayor que el calor generado la temperatura seguira disminuyendo hasta llegar al estado A.
por tanto como la pendiente de la entalpia del calor  ganado es mayor que la del calor eliminado, es inestable ya que cualquier perturbación lo lleva a otro estado estacionario.

\subsection{Bateria de tanques en serie}
\begin{flushleft}
	En una batería de $N$ tanques en serie, el volumen total $V$ se reparte en $N$ tanques de volumen $v_i$. El tiempo espacial de cada tanque es:
	\[
		\tau_i = \frac{v_i}{F}
	\]
	y el tiempo espacial global:
	\[
		\tau = N\,\tau_i
	\]

	Cada tanque recibe como alimentación la corriente de salida del anterior. Así:
	\begin{itemize}
		\item Tanque 1: entrada con conversión $x_{A0}$, salida $x_{A1}$.
		\item Tanque 2: entrada $x_{A1}$, salida $x_{A2}$.
		\item $\dots$
		\item Tanque $n$: entrada $x_{A(n-1)}$, salida $x_{An}$ (o $x_{As}$).
	\end{itemize}

	El caudal $F$ se mantiene constante en todos los tanques, mientras que la conversión aumenta progresivamente a lo largo de la serie, de acuerdo con la ecuación de diseño (ec. 3.10).
\end{flushleft}
\begin{equation}
	\tau_i = \frac{v_i}{F} \rightarrow \tau = \frac{V}{F} = n\cdot \frac{v_i}{F} = n\cdot \tau_i
\end{equation}

\begin{equation}
	\frac{\tau_i}{C_{AR}} = \frac{x_{A1}-x_{A0}}{(-r_{A})_1} = \frac{x_{A2}-x_{A1}}{(-r_{A})_2} = \dots = \frac{x_{An}-x_{A(n-1)}}{(-r_{A})_n}
\end{equation}
\vspace{1em}
\noindent se nos pueden presentar dos casos fundamentalmente:
\begin{itemize}
	\item que conozcamos el numero de tanques con el que vayamos a trabajar es decir N
	\item lo que conocemos es la conversión a la entrada $x_{AO}$ queremos una determinada conversion a la salida $x_{An}$
\end{itemize}
\begin{flushleft}
	Para el cálculo en una batería de $N$ tanques en serie, se puede resolver el problema de forma secuencial. Conocido el tiempo espacial $\tau_i$ y la conversión de entrada, se calcula la conversión a la salida de cada tanque de manera sucesiva: $x_{A1}$, $x_{A2}$, ..., hasta $x_{AN}$.

	En diseño (segundo caso), se suele suponer un número de tanques $N$ y un $\tau_i$, y realizar el cálculo en sentido inverso: partiendo de la conversión de salida conocida, se determinan las conversiones intermedias hasta obtener la conversión de entrada $x_{A0}$.

	Si la conversión calculada $x_{A0}$ no coincide con la requerida:
	\begin{itemize}
		\item Si $x_{A0}$ calculada es mayor, faltan tanques $\Rightarrow$ aumentar $N$.
		\item Si $x_{A0}$ calculada es menor, sobran tanques $\Rightarrow$ disminuir $N$.
	\end{itemize}

	Se itera hasta que la conversión calculada coincide con la especificada.
\end{flushleft}

\mlenma{Ventajas y desventajas de la batería de tanques}{
	La principal ventaja es que genera un perfil de concentraciones: cada reactor trabaja a mayor concentración que el siguiente, lo que implica mayores velocidades de reacción y, por tanto, mayor rendimiento global.

	Para un mismo volumen total:
	\begin{itemize}
		\item 1 tanque $\Rightarrow$ menor conversión.
		\item Varios tanques $\Rightarrow$ mayor conversión.
		\item $N \to \infty$ $\Rightarrow$ comportamiento de reactor flujo pistón.
	\end{itemize}

	Como desventaja, requiere más equipos, mayor complejidad de operación y mayor coste.

	Además, permite operar distintas etapas en condiciones diferentes (temperatura, pH), lo cual es útil en procesos con múltiples reacciones.
}

\qs{RCMP-1}{Un reactor continuo mezcla perfecta isotermo de 15 L es alimentado por una corriente de 1 L/h a una concentración de 1 mol/L del compuesto A. En el reactor se lleva a cabo la reacción química:

	A $\rightarrow$ B\\
	Constante de velocidad = 1 h$^{-1}$

	\begin{enumerate}[label={\alph*)}]
		\item Obtener las condiciones del estado estacionario
		\item Obtener la dinámica del proceso si el reactor se pone en marcha en las mismas condiciones que la alimentación
	\end{enumerate}}
\begin{minted}[fontsize=\small, linenos]{scilab}
clear; clc; 
// RCMP-1.sce
// A => B
// Isotermo

// (a) Estado estacionario

// SISTEMA DE ECUACIONES ALGEBRAICAS
function dxdt = f(x)
    // Variables
    CA = x(1)
    CB = x(2)
    // Velocidad de reacción
    r = k*CA
    // Balance de materia para A
    // d(V*CA)dt = F*CA0 - F*CA - r*V
    dCAdt = F*(CA0-CA)/V - r
    // Balance de materia para B
    // d(V*CB)dt = F*CB0 - F*CB + r*V
    dCBdt = F*(CB0-CB)/V + r
    // Derivadas
    dxdt(1) = dCAdt
    dxdt(2) = dCBdt
endfunction

	\end{enumerate}}

clear; clc; 
// RCMP-2.sce
// A + B <=> C
// Isotermo

// (a) Estado estacionario

// SISTEMA DE ECUACIONES ALGEBRAICAS
function dxdt = f(x)
    // Variables
    CA = x(1)
    CB = x(2)
    CC = x(3)
    // Velocidad de reacción
    // r = rd - ri = kd*CA*CB - ki*CC = kd*CA*CB - kd*CC/Keq
    r = kd*(CA*CB - CC/Keq)
    // Balance de materia para A
    // d(V*CA)dt = F*CA0 - F*CA - r*V
    dCAdt = F*(CA0-CA)/V - r
    // Balance de materia para B

    // Balance de materia para C
    // d(V*CC)dt = F*CC0 - F*CC + r*V
    dCCdt = F*(CC0-CC)/V + r
    // Derivadas

    dxdt(2) = dCBdt
    dxdt(3) = dCCdt
endfunction

clear; clc; 
// RCMP-3.sce
// A => B
// No adiabático

// (a) Método. gráfico

// CONSTANTES
CP = 0.8; // cal/(g*K)
RHO = 1000; // g/L
F = 20; // L/s
V = 1500; // L
UA = 1E4; // cal/(K*s)
H = -8E4; // cal/mol
T0 = 293; // K
TJ = 283; // K
CA0 = 2.5; // mol/L
k0 = 2.5E10; // 1/s
R = 1.987; // cal/(mol*K)
E = 2.1E4; // cal/mol

// BALANCES 

scf(1); clf(1);
xlabel('CA'); CAmin = 0; CAmax = 3;
ylabel('T'); Tmin = 280; Tmax = 500;

dT = 0.1; T = Tmin:dT:Tmax;
k = k0*exp(-E./(R*T)); // Ecuación de Arrhenius
Q = UA*(T-TJ); // Calor transferido del reactor a la camisa

// Balance de materia para A
// d(V*CA)dt = F*CA0 - F*CA - r*V = F*CA0 - F*CA - k*CA*V = 0
// F*CA0 = CA * (F + k*V)
CAbm = F*CA0./(F+k*V);
plot(CAbm,T,'r-');

// Balance de energía
// d(V*RHO*CP*T)dt = F*RHO*CP*T0 - F*RHO*CP*T -H*r*V - Q = 0
// F*RHO*CP*(T0-T) - Q = H*k*CA*V
CAbe = (F*RHO*CP*(T0-T) - Q)./(H*k*V); 
plot(CAbe,T,'r--');

// LOCALIZACIÓN DE ESTADOS ESTACIONARIOS
y = CAbm-CAbe;
indexy0 = find(y(1:$-1).*y(2:$)<0)+1;

Teeg = T(indexy0)

// (b) Estado estacionario

// SISTEMA DE ECUACIONES ALGEBRAICAS
function dxdt = f(x)
    // Variables
    CA = x(1)
    T  = x(2)
    // Ecuación de Arrhenius
    k = k0*exp(-E/(R*T))
    // Velocidad de reacción
    r = k*CA
    // Calor transferido del reactor a la camisa
    Q = UA*(T-TJ)
    // Balance de materia para A
    //d(V*CA)dt = F*CA0 - F*CA - r*V
    dCAdt = F*(CA0-CA)/V - r
    // Balance de energía
    // d(V*RHO*CP*T)dt = F*RHO*CP*T0 - F*RHO*CP*T -H*r*V - Q
    dTdt = F*(T0-T)/V - H*r/(RHO*CP) - Q/(V*RHO*CP) 
    // Derivadas
    dxdt(1) = dCAdt
    dxdt(2) = dTdt
endfunction

Teeguess = 293; // K

// RESOLVER

    dxdt = f(x)
endfunction

\end{minted}
